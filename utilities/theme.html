# Theming

affinity-css supports Light/Dark Theme.

Add the Attribute data-theme : theme-dark for Dark Mode.

Add this script for toggling.

const html = document.querySelector('html');
html.dataset.theme = `theme-dark`;
let isDark = true;
function switchTheme() {
  isDark = !isDark;
  if (isDark) {
    html.dataset.theme = `theme-dark`;
  } else {
    html.dataset.theme = `theme-light`;
  }
}
