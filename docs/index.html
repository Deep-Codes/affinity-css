const fs = require('fs');
    const data = fs.readFileSync('./colors/_palette.scss', 'utf-8');
    const arr = data.split('\n');
    let newArr = [];
    arr.forEach((el) => {
      const indexSt = el.indexOf('-');
      const indexEnd = el.indexOf(':');
      let temp = el.slice(indexSt + 1, indexEnd);
      if (indexSt >= 0) {
        let num = +el.slice(indexSt + 1, indexEnd);
        if (num === 50) {
          num = '';
        } else {
          num = num / 100;
        }
        // console.log(num);
        let newStr = el.replace(temp, num.toString());
        newStr.replaceAll('"', '');
        newArr.push(newStr.slice(0, -1));
      }
    });
    fs.writeFileSync('./colors/_tailwind.scss', JSON.stringify(newArr));
